// 生成CSV内容
function generateCSV(users) {
  const headers = ['ID', '昵称', '真实姓名', '手机号', '状态', '会员等级', '服务次数', '总消费', '注册时间'];
  const rows = users.map(user => [
    user.id,
    user.nickname,
    user.real_name || '',
    user.phone || '',
    user.status,
    user.member_level,
    user.service_count,
    user.total_spent,
    Utils.formatDate(user.register_time)
  ]);
  
  const csvContent = [headers, ...rows]
    .map(row => row.map(field => `"${field}"`).join(','))
    .join('\n');
  
  return csvContent;
}
